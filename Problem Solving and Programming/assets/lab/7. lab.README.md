### Write a C++ program to show the

#### Matrix concept

<img src="../images/matrix.png" width="400">

```cpp
#include <iostream>
using namespace std;

int main()
{
    // Creating a 3x3 integer matrix
    int a[3][3];

    // Asking the user to enter 9 elements
    cout << "Enter 9 elements for the 3x3 Matrix: ";

    // Taking input from the user (row-wise)
    for (int i = 0; i < 3; i++)
    {
        for (int j = 0; j < 3; j++)
        {
            cin >> a[i][j];   // Storing element at row i and column j
        }
    }

    // Printing the matrix in proper 3x3 format
    cout << "\nThe 3x3 Matrix is:\n";
    for (int i = 0; i < 3; i++)
    {
        // Printing one row at a time
        for (int j = 0; j < 3; j++)
        {
            cout << a[i][j] << " ";   // Printing element with space
        }
        cout << endl; // Move to next line after printing each row
    }

    return 0; // Program ends
}
```

#### a. Matrix addition of 3x3 matrix

```cpp
#include <iostream>
using namespace std;

int main()
{
    // Declaring two 3x3 matrices and one result matrix
    int A[3][3], B[3][3], C[3][3];

    // Input for Matrix A
    cout << "Enter 9 elements for Matrix A:\n";
    for (int row = 0; row < 3; row++)
    {
        for (int col = 0; col < 3; col++)
        {
            cin >> A[row][col];   // Storing each element in Matrix A
        }
    }

    // Input for Matrix B
    cout << "\nEnter 9 elements for Matrix B:\n";
    for (int row = 0; row < 3; row++)
    {
        for (int col = 0; col < 3; col++)
        {
            cin >> B[row][col];   // Storing each element in Matrix B
        }
    }

    // Adding A and B into C
    for (int row = 0; row < 3; row++)
    {
        for (int col = 0; col < 3; col++)
        {
            C[row][col] = A[row][col] + B[row][col];   // Element-wise addition
        }
    }

    // Displaying the result matrix
    cout << "\nResult of Matrix Addition (A + B):\n";
    for (int row = 0; row < 3; row++)
    {
        for (int col = 0; col < 3; col++)
        {
            cout << C[row][col] << " ";   // Printing the element
        }
        cout << endl;   // Move to next line after each row
    }

    return 0;
}
```
