# VS Code

```c++
/*******************************************
Write a C++ program to demonstrate
b. Temperature converter (Celsius ↔ Fahrenheit).
Programmer - Dherya Agrawal, CSE-Cloud, 1 Sem
Written For - Karnavati University
********************************************/

#include <iostream>   // For input and output
using namespace std;


int main() {
    int choice;        // To store user's choice (1 or 2)
    double temp;       // To store input temperature
    double result;     // To store converted temperature


    // Display menu
    cout << "=======================" << endl;
    cout << "   Temperature Converter" << endl;
    cout << "=======================" << endl;


    cout << "1. Celsius to Fahrenheit" << endl;   // Option 1
    cout << "2. Fahrenheit to Celsius" << endl;   // Option 2
    cout << "Enter your choice (1 or 2): ";       // Ask for user input
    cin >> choice;                                // Take input


    // If user chooses Celsius to Fahrenheit
    if (choice == 1) {
        cout << "Enter temperature in Celsius: ";
        cin >> temp;   // Input Celsius value
        result = (temp * 9.0 / 5.0) + 32;   // Formula: (°C × 9/5) + 32 = °F
        cout << "Temperature in Fahrenheit: " << result << endl;
    }
    // If user chooses Fahrenheit to Celsius
    else if (choice == 2) {
        cout << "Enter temperature in Fahrenheit: ";
        cin >> temp;   // Input Fahrenheit value
        result = (temp - 32) * 5.0 / 9.0;   // Formula: (°F - 32) × 5/9 = °C
        cout << "Temperature in Celsius: " << result << endl;
    }
    // If user enters invalid choice
    else {
        cout << "Invalid choice!" << endl;
    }


    return 0;   // End of program
}
```
