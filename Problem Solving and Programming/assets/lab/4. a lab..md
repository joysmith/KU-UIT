```c++
/*******************************************
Write a C++ program to demonstrate
a. Write a C++ program to calculate the factorial using: a. For loop, while loop, and do-while loop
Programmer - Shukla Aditya Santonsh, CSE-Cloud, 1 Sem
Written For - Karnavati University
********************************************/


#include<iostream.h>
#include<conio.h>

int main()
{
    clrscr();

    // Variable for users input.
    int number;
    int enteredNumber;

    cout << "Enter your number you want factorial of: ";
    cin >> number;

    enteredNumber = number;

    // variable to store the result value.
    double result = 1;

    // Creating a loop for recursive multiplication.
    while(number>0)
    {
        result = result * number;
        number--;
    }

    // to print the result.
    cout << "Factorial of " << enteredNumber << " is: " << result << endl;


    getch();

    return 0;
}

/*
OUTPUT:
Enter your number you want factorial of: 5
Factorial of 5 is: 120
*/
```

### Using for Loop

```cpp
/*******************************************
Write a C++ program to demonstrate
a. Write a C++ program to calculate the factorial using: a. For loop, while loop, and do-while loop
Programmer - Neil Thakkar, CSE-Cloud, 1 Sem
Written For - Karnavati University
********************************************/


#include<iostream.h>
#include<conio.h> //this library need to be use in turbo for input and output

int main() {
  	clrscr();

    int number, factorial = 1;

    cout << "Enter a non-negative number: ";
    cin >> number;

    // Calculate factorial using for loop
    for (int i = 1; i <= number; i++) {
        factorial = factorial * i;
    }

    cout << "Factorial of " << number << " = " << factorial << endl;

    getch();
    return 0;
}

/*
OUTPUT:
Enter your number you want factorial of: 5
Factorial of 5 is: 120
*/
```
